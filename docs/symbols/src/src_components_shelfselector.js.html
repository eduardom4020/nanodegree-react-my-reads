1 import React from 'react'
  2 
  3 /**
  4 * @class
  5 * @classdesc This class describes the shelves selector.
  6 * @type {Component}
  7 */
  8 const ShelfSelector = ({shelves, moveBookToShelf=null, bookAttached=null}) => (
  9     <select 
 10         onChange={ event => (
 11             bookAttached 
 12             && moveBookToShelf 
 13             && moveBookToShelf(bookAttached, event.target.value)
 14         )}
 15         defaultValue={bookAttached.shelf}
 16     >
 17         <option value='move' disabled key={'shelfSelectorMoveTo'}>Move to...</option>
 18         {
 19             shelves.map((shelf, it) => (
 20                 <option 
 21                     value={shelf.value}
 22                     key={`shelfSelectorOption${it}`}
 23                 >
 24                     {shelf.name}
 25                 </option>
 26             ))
 27         }
 28         <option value='none' key={'shelfSelectorNone'}>None</option>
 29     </select>
 30 );
 31 
 32 export default ShelfSelector;
