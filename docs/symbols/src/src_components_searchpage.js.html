 36 /**
 37  * Search Page
 38  * @namespace SearchPage
 39  */
 40 
 41 /**
 42 * @description The initial configuration of search shelves. 
 43 * These are not user workspace shelfs and will not have durable information.
 44 * @constant
 45 * @type {Array}
 46 * @default
 47 */
 48 const INITIAL_SEARCH_SHELVES = [
 49     {
 50         value: 'SEARCH__result',
 51         name: 'Result'
 52     }
 53 ];
 54 
 55 /**
 56 * @class
 57 * @classdesc This class describes the search page. It contains one or more search shelves.
 58 * The books in it's state will be a combination of books that user has in shelf that matchs
 59 * the search result and books that are not in shelves but match even.
 60 * @type {Component}
 61 */
 62 class SearchPage extends Component {
 63     constructor(props) {
 64         super(props);
 65 
 66         this.state = {
 67             books: [],
 68             shelves: INITIAL_SEARCH_SHELVES,
 69             searchText: '',
 70         };
 71     }
 72 
 73     /**
 74     * @description Every time a user writes on input field, this method is called.
 75     * It sets up a timeout to trigger another method, but if this method is called
 76     * again, this timeout restart. When user stops writing, the next method is triggered.
 77 	* @param {Object} event
 78 	* @returns {null} This method returns nothing.
 79 	*/
 80     dynamicSearch = event => {
 81         const value = event.target.value;
 82 
 83         if(this.timer) {
 84             clearTimeout(this.timer);
 85         }
 86 
 87         this.timer = setTimeout(this.execSearch(value) , 500);
 88         this.setState({
 89             searchText: value
 90         })
 91     }
 92 
 93     /**
 94     * @description Use the BooksAPI to search all books that match some search term, them set books state.
 95 	* @param {string} value
 96 	* @returns {null} This method returns nothing.
 97 	*/
 98     execSearch = value => () => {
 99         if(value === '') {
100             this.setState({
101                 books: []
102             })
103         } else {
104             BooksAPI.search(value).then(res => {
105                 const booksTreated = this.treatSearchResult(res);
106 
107                 this.setState({
108                     books: booksTreated
109                 });
110             });
111         }
112     }
113 
114     /**
115     * @description Get all books resulting of a search and check if they exists on user's shelves.
116     * For them, change the default shelf (None) to the shelf this book is currently inside.
117 	* @param {?Array} queryRes
118 	* @returns {Array} This method returns books of search with some changes on shelves if needed.
119 	*/
120     treatSearchResult = queryRes => {
121         if(queryRes.constructor !== Array) {
122             return [];
123         }   
124 
125         const {books: shelvesBooks} = this.props;
126 
127         const booksRes = queryRes.map(book => {
128             const bookFromShelfFilter = shelvesBooks.filter(shelfBook => shelfBook.id === book.id);
129             if(bookFromShelfFilter.length > 0) {
130                 return bookFromShelfFilter[0];
131             } else {
132                 return book;
133             }
134         });
135 
136         return booksRes;
137     }
138     
139     /**
140     * @description Create a global timer object. It will be used as search timeout.
141 	*/
142     componentDidMount() {
143         this.timer = null;
144     }
145 
146     /**
147     * @description The render method. It contains all of the search tools and a path back
148     * to main page. It contains too a list of shelves, that are shelves that contains query results.
149 	*/
150     render() {
151         return (
152             <div className='search-books'>
153                 <div className='search-books-bar'>
154                     <Link to='/'>
155                         <button className='close-search'>Close</button>
156                     </Link>
157                     <div className='search-books-input-wrapper'>
158                         <input 
159                             type='text'
160                             value={this.state.searchText}
161                             placeholder='Search by title or author'
162                             onChange={this.dynamicSearch}
163                         />
164                     </div>
165                 </div>
166                 <div className='search-books-results'>
167                 {
168                     this.state.shelves.map(shelf => (
169                         <Shelf 
170                             {...this.props}
171                             books={this.state.books}
172                             thisShelf={shelf}
173                             filterFunc={book => book}
174                         >
175                             <div className='empty-list-books'>
176                                 <h3>Search Results Nothing</h3>
177                                 <p>Please make a search using the restricted terms of API using the input above.</p>
178                             </div>
179                         </Shelf>
180                     ))
181                 }
182                 </div>
