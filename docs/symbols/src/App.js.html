 51 /**
 52 * @class
 53 * @classdesc This is the main class that describe the component that cointains the entire
 54 * application. It state contains the info necessary for the application works well. By passing
 55 * down to children these informations and the methods to change then, this class assures that
 56 * all of the information remains consistent at the two pages.
 57 * @type {Component}
 58 */
 59 class BooksApp extends Component {
 60 	constructor(props) {
 61 		super(props);
 62 
 63 		this.state = {
 64 			books: [],
 65 			shelves: INITIAL_SHELVES,
 66 			mounted: false
 67 		};
 68 	}
 69 
 70 	/**
 71 	* @description Receives a list of books object (it's only necessary that these objects
 72 	* have an valid ID on BooksAPI) and sets the state of books in BooksApp component.
 73 	* @param {Array.<Book>} books
 74 	* @returns {null} This method returns nothing.
 75 	*/
 76 	setBooks = books => {
 77 		this.setState({
 78 			books: books
 79 		});
 80 	}
 81 
 82 	/**
 83 	* @description Receives a list of books object (it's only necessary that these objects
 84 	* have an valid ID on BooksAPI) and compares with books currently in BooksApp component
 85 	* state.
 86 	* @param {Array.<Book>} toCompareBooks
 87 	* @returns {Boolean} Returns true if any param or BooksApp books state are null. Also
 88 	* returns true if param does not match with state, otherwise returns false.
 89 	*/
 90 	booksChanged = toCompareBooks => {
 91 		if(toCompareBooks && this.state.books) {
 92 			const nextStateIdShelf = toCompareBooks.map(book => ({[book.id]: book.shelf}));
 93 			const currStateIdShelf = this.state.books.map(book => ({[book.id]: book.shelf}));
 94 			
 95 			return !ArrayOfObjectsEquality(currStateIdShelf, nextStateIdShelf);
 96 		} else {
 97 			return true;
 98 		}
 99 	}
100 
101 	/**
102 	* @description Use BooksAPI to get all books on user's shelves. Then sets the BooksApp state
103 	* with these books.
104 	* @returns {null} This method returns nothing.
105 	*/
106 	refreshBooks = () => {
107         BooksAPI.getAll()
108 		.then(books => {
109 			this.setBooks(books);
110 		});
111     }
112 
113 	/**
114 	* @description Use BooksAPI to move book to none shelf. Then sets the BooksApp state
115 	* to exclude this book.
116 	* @param {Book} book
117 	* @returns {null} This method returns nothing.
118 	*/
119 	removeBookFromShelf = book => {
120 		BooksAPI.update(book, 'none')
121 		.then(this.setState(prevState => ({
122 			books: prevState.books.filter(oldStateBook => oldStateBook.id !== book.id)
123 		})))
124 	}
125 
126 	/**
127 	* @description Use removeBookFromShelf to remove all books from all shelves.
128 	* @returns {null} This method returns nothing.
129 	*/
130 	clearAllShelves = () => {
131 		this.state.books.forEach(book => {
132 			this.removeBookFromShelf(book);
133 		});
134 	}
135 
136 	/**
137 	* @description Get shelf value by passing an index of shelves Array in BooksApp state or a string
138 	* that is the name of the shelf.
139 	* @param {(number | string)} shelf
140 	* @returns {(string | Boolean)} This method returns an string that is the value of that shelf or
141 	* a boolean always false, that represents that the input name does not exists on shelves.
142 	*/
143 	validateAndGetShelfValue = shelf => {
144 		if(shelf === 'none') {
145 			return 'none';
146 		} else if(typeof shelf == 'number') {
147 			return this.state.shelves[shelf].value;
148 		} else if(typeof shelf == 'string') { 
149 			const filterShelves = this.state.shelves.filter(aShelf => aShelf.value === shelf);
150 			return filterShelves.length > 0 ? shelf : false;
151 		} else {
152 			return false;
153 		}
154 	} 
155 
156 	/**
157 	* @description Use BooksAPI to move a book to some shelf. Then update BooksApp state.
158 	* @param {Book} book
159 	* @param {(number | string)} shelf
160 	* @returns {null} This method returns nothing.
161 	*/
162 	moveBookToShelf = (book, shelf) => {
163 		const shelfValue = this.validateAndGetShelfValue(shelf);
164 		
165 		if(shelfValue) { 
166 			BooksAPI.update(book, shelfValue)
167 			.then(this.setState(prevState => {
168 					const booksNewState = prevState.books.map(aBook => aBook.id === book.id ? {...aBook, shelf: shelfValue} : aBook);
169 					return ({
170 						books: booksNewState
171 					})
172 				})
173 			)	
174 		}
175 	}
176 
177 	/**
178 	* @description Use BooksAPI to update a group of books info changing them to some shelf.
179 	* @param {Array.<Object.<Book, number>>} booksPerShelf
180 	* @returns {Array.<Promise>} This method returns an array of Promises containing the new state of
181 	* the moved books.
182 	*/
183 	updateGroupOfBooksAPI = booksPerShelf => {
184 		const result = booksPerShelf.map(async bookToShelf => {
185 			const shelfValue = this.state.shelves[bookToShelf.toShelf].value;
186 			await BooksAPI.update(bookToShelf.book, shelfValue);
187 			return {...bookToShelf.book, shelf: shelfValue};
188 		});
189 		return result;
190 	}
191 
192 	/**
193 	* @description Use BooksAPI to move a group of books to some shelf. Then update BooksApp state.
194 	* @param {Book} book
195 	* @param {(number | string)} shelf
196 	* @returns {null} This method returns nothing.
197 	*/
198 	moveBooksToShelfByOrder = (books, shelf) => {
199 		const bookToShelf = books.map(book => ({book: book, toShelf: shelf}));
200 		const res = this.updateGroupOfBooksAPI(bookToShelf);
201 		Promise.all(res)
202 		.then(newBooksState => {
203 			this.setState({
204 				books: newBooksState
205 			});
206 		});
207 	}

233 	* @description Refresh books info and sets mounted state to true, that will prevent Main Page
234 	* to execute refreshBooks method on initial load.
235 	* @returns {null} This method returns nothing.
236 	*/
237 	componentDidMount() {
238 		this.refreshBooks();
239 		this.setState({
240 			mounted: true
241 		});
242 	}

244 	/**
245 	* @description BooksApp's render. It contains the two pages, uses react router to 
246 	* render them and the float buttons. The functions and the state are passed down
247 	* to childrens in this method.
248 	*/
249 	render() {
250 		return (
251 			<div className='app'>
252 				<Route exact path='/' render={() => (
253 						<MainPage 
254 							{...this.state} 
255 							setBooks={this.setBooks}
256 							booksChanged={this.booksChanged}
257 							moveBookToShelf={this.moveBookToShelf}
258 							refreshBooks={this.state.mounted ? this.refreshBooks : null}
259 						/>
260 					)} 
261 				/>
262 
263 				<Route exact path='/search' render={() => (
264 						<SearchPage 
265 							{...this.state} 
266 							moveBookToShelf={this.moveBookToShelf}
267 						/>
268 					)} 
269 				/>
270 
271 				{
272 					ADD_BUTTON_PATHS.map((buttonPath, it) => (
273 						<Route exact path={buttonPath} component={AddBookButton} key={`btPath${it}`}/>
274 					))
275 				}
276 			</div>
277 		);
278 	}
279 }
